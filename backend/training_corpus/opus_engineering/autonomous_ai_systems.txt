@dataclass
class HealthMetric:
    name: str
    value: float
    threshold_warning: float
    threshold_critical: float
    timestamp: datetime = field(default_factory=datetime.now)
    
    @property
    def status(self) -> HealthStatus:
        if self.value >= self.threshold_critical:
            return HealthStatus.CRITICAL
        elif self.value >= self.threshold_warning:
            return HealthStatus.DEGRADED

class AnomalyDetector:
    def __init__(self, contamination: float = 0.1):
        self.isolation_forest = IsolationForest(contamination=contamination, random_state=42)
        self.scaler = StandardScaler()
        self.is_trained = False
        self.feature_history = []
    
    def train(self, historical_data: List[List[float]]):
        """Train the anomaly detector with historical normal data"""
        if len(historical_data) < 10:
            return False
        
        scaled_data = self.scaler.fit_transform(historical_data)
        self.isolation_forest.fit(scaled_data)
        self.is_trained = True
        return True
    
    def detect_anomaly(self, features: List[float]) -> Tuple[bool, float]:
        """Detect if current features represent an anomaly"""
        if not self.is_trained:
            return False, 0.0
        
        scaled_features = self.scaler.transform([features])
        anomaly_score = self.isolation_forest.decision_function(scaled_features)[0]
        is_anomaly = self.isolation_forest.predict(scaled_features)[0] == -1
        
        return is_anomaly, anomaly_score

class DiagnosticEngine:
    def __init__(self):
        self.diagnostic_rules = {}
        self.setup_default_rules()
    
    def setup_default_rules(self):
        """Setup default diagnostic rules"""
        self.diagnostic_rules = {
            "high_cpu_usage": {
                "condition": lambda metrics: any(m.name == "cpu_usage" and m.value > 90 for m in metrics),
                "diagnosis": "High CPU usage detected",
                "potential_causes": ["Resource-intensive process", "Memory leak", "Infinite loop"],
                "recommended_actions": ["restart_service", "kill_high_cpu_processes", "scale_resources"]
            },
            "high_memory_usage": {
                "condition": lambda metrics: any(m.name == "memory_usage" and m.value > 95 for m in metrics),
                "diagnosis": "High memory usage detected",
                "potential_causes": ["Memory leak", "Large dataset processing", "Insufficient memory allocation"],
                "recommended_actions": ["restart_service", "garbage_collect", "scale_memory"]
            },
            "high_error_rate": {
                "condition": lambda metrics: any(m.name == "error_rate" and m.value > 15 for m in metrics),
                "diagnosis": "High error rate in AI model",
                "potential_causes": ["Model degradation", "Input data quality issues", "Service dependency failure"],
                "recommended_actions": ["rollback_model", "validate_input_data", "check_dependencies"]
            }
        }
    
    def diagnose(self, metrics: List[HealthMetric], component: str) -> Dict[str, Any]:
        """Diagnose issues based on metrics"""
        diagnosis = {
            "component": component,
            "issues_found": [],
            "recommended_actions": [],
            "severity": IncidentSeverity.LOW
        }
        
        max_severity = IncidentSeverity.LOW
        
        for rule_name, rule in self.diagnostic_rules.items():
            if rule["condition"](metrics):
                diagnosis["issues_found"].append({
                    "rule": rule_name,
                    "diagnosis": rule["diagnosis"],
                    "potential_causes": rule["potential_causes"]
                })
                diagnosis["recommended_actions"].extend(rule["recommended_actions"])
                
                # Determine severity based on metric values
                critical_metrics = [m for m in metrics if m.status == HealthStatus.CRITICAL]
                if critical_metrics:
                    max_severity = IncidentSeverity.CRITICAL
                elif any(m.status == HealthStatus.DEGRADED for m in metrics):
                    max_severity = max(max_severity, IncidentSeverity.MEDIUM)
        
        diagnosis["severity"] = max_severity
        return diagnosis

class LearningEngine:
    def __init__(self, knowledge_base_path: str = "self_healing_knowledge.json"):
        self.knowledge_base_path = Path(knowledge_base_path)
        self.knowledge_base = self._load_knowledge_base()
        self.incident_history = []
    
    def _load_knowledge_base(self) -> Dict[str, Any]:
        """Load existing knowledge base"""
        if self.knowledge_base_path.exists():
            with open(self.knowledge_base_path, 'r') as f:
                return json.load(f)
        return {
            "successful_fixes": {},
            "failed_fixes": {},
            "patterns": {},
            "effectiveness_scores": {}
        }
    
    def _save_knowledge_base(self):
        """Save knowledge base to disk"""
        with open(self.knowledge_base_path, 'w') as f:
            json.dump(self.knowledge_base, f, indent=2, default=str)
    
    def learn_from_incident(self, incident: Incident):
        """Learn from an incident and its resolution"""
        self.incident_history.append(incident)
        
        if incident.resolved_at and incident.fix_applied:
            # Record successful fix
            fix_key = f"{incident.component}_{incident.fix_applied}"
            
            if incident.verification_result:
                # Successful fix
                if fix_key not in self.knowledge_base["successful_fixes"]:
                    self.knowledge_base["successful_fixes"][fix_key] = []
                
                self.knowledge_base["successful_fixes"][fix_key].append({
                    "incident_id": incident.id,
                    "severity": incident.severity.value,
                    "description": incident.description,
                    "resolution_time": (incident.resolved_at - incident.detected_at).total_seconds(),
                    "timestamp": incident.resolved_at.isoformat()
                })
                
                # Update effectiveness score
                self._update_effectiveness_score(fix_key, True)
            else:
                # Failed fix
                if fix_key not in self.knowledge_base["failed_fixes"]:
                    self.knowledge_base["failed_fixes"][fix_key] = []
                
                self.knowledge_base["failed_fixes"][fix_key].append({
                    "incident_id": incident.id,
                    "severity": incident.severity.value,
                    "description": incident.description,
                    "timestamp": incident.detected_at.isoformat()
                })
                
                self._update_effectiveness_score(fix_key, False)
        
        # Detect patterns
        self._detect_patterns()
        
        # Save updated knowledge base
        self._save_knowledge_base()
    
    def _update_effectiveness_score(self, fix_key: str, success: bool):
        """Update effectiveness score for a fix"""
        if fix_key not in self.knowledge_base["effectiveness_scores"]:
            self.knowledge_base["effectiveness_scores"][fix_key] = {
                "successes": 0,
                "failures": 0,
                "score": 0.0
            }
        
        scores = self.knowledge_base["effectiveness_scores"][fix_key]
        if success:
            scores["successes"] += 1
        else:
            scores["failures"] += 1
        
        total = scores["successes"] + scores["failures"]
        scores["score"] = scores["successes"] / total if total > 0 else 0.0
    
    def _detect_patterns(self):
        """Detect patterns in incident history"""
        if len(self.incident_history) < 5:
            return
        
        # Pattern: Recurring incidents in the same component
        component_incidents = {}
        for incident in self.incident_history[-20:]:  # Last 20 incidents
            if incident.component not in component_incidents:
                component_incidents[incident.component] = []
            component_incidents[incident.component].append(incident)
        
        for component, incidents in component_incidents.items():
            if len(incidents) >= 3:
                # Check if incidents are recurring (within 24 hours)
                recent_incidents = [i for i in incidents 
                                 if (datetime.now() - i.detected_at).total_seconds() < 86400]
                
                if len(recent_incidents) >= 2:
                    pattern_key = f"recurring_{component}"
                    self.knowledge_base["patterns"][pattern_key] = {
                        "type": "recurring_incidents",
                        "component": component,
                        "frequency": len(recent_incidents),
                        "last_occurrence": max(i.detected_at for i in recent_incidents).isoformat(),
                        "recommendation": "Consider proactive monitoring or architectural changes"
                    }
    
    def get_recommended_fix(self, component: str, issue_type: str) -> Optional[str]:
        """Get recommended fix based on historical data"""
        # Find the most effective fix for similar issues
        best_fix = None
        best_score = 0.0
        
        for fix_key, score_data in self.knowledge_base["effectiveness_scores"].items():
            if component in fix_key and score_data["score"] > best_score:
                best_fix = fix_key.split('_', 1)[1]  # Remove component prefix
                best_score = score_data["score"]
        
        return best_fix if best_score > 0.5 else None

class SelfHealingSystem:
    def __init__(self):
        self.monitors: List[HealthMonitor] = []
        self.anomaly_detector = AnomalyDetector()
        self.diagnostic_engine = DiagnosticEngine()
        self.autofix_engine = AutoFixEngine()
        self.verification_engine = VerificationEngine()
        self.learning_engine = LearningEngine()
        
        self.active_incidents: Dict[str, Incident] = {}
        self.monitoring_interval = 30  # seconds
        self.is_running = False
        self.logger = logging.getLogger(__name__)
        
        # Setup logging
        logging.basicConfig(level=logging.INFO)
    
    def add_monitor(self, monitor: HealthMonitor):
        """Add a health monitor"""
        self.monitors.append(monitor)
    
    async def start(self):
        """Start the self-healing system"""
        self.is_running = True
        self.logger.info("Self-healing system started")
        
        # Start monitoring loop
        monitoring_task = asyncio.create_task(self._monitoring_loop())
        
        # Train anomaly detector with historical data if available
        await self._train_anomaly_detector()
        
        await monitoring_task
    
    def stop(self):
        """Stop the self-healing system"""
        self.is_running = False
        self.logger.info("Self-healing system stopped")
    
    async def _monitoring_loop(self):
        """Main monitoring loop"""
        while self.is_running:
            try:
                await self._monitor_and_heal()
                await asyncio.sleep(self.monitoring_interval)
            except Exception as e:
                self.logger.error(f"Error in monitoring loop: {e}")
                await asyncio.sleep(5)  # Short delay before retrying
    
    async def _monitor_and_heal(self):
        """Monitor system health and trigger healing if needed"""
        for monitor in self.monitors:
            try:
                # Collect metrics
                metrics = await monitor.collect_metrics()
                component = monitor.get_component_name()
                
                # Check for anomalies
                feature_vector = [m.value for m in metrics]
                is_anomaly, anomaly_score = self.anomaly_detector.detect_anomaly(feature_vector)
                
                # Check for critical metrics or anomalies
                critical_metrics = [m for m in metrics if m.status == HealthStatus.CRITICAL]
                degraded_metrics = [m for m in metrics if m.status == HealthStatus.DEGRADED]
                
                if critical_metrics or (is_anomaly and anomaly_score < -0.5):
                    # Create or update incident
                    incident_id = f"{component}_{int(time.time())}"
                    
                    if incident_id not in self.active_incidents:
                        # Diagnose the issue
                        diagnosis = self.diagnostic_engine.diagnose(metrics, component)
                        
                        # Create incident
                        incident = Incident(
                            id=incident_id,
                            component=component,
                            severity=diagnosis["severity"],
                            description=f"Health issues detected: {diagnosis['issues_found']}",
                            detected_at=datetime.now(),
                            metadata={
                                "metrics": [{"name": m.name, "value": m.value, "status": m.status.value} 
                                          for m in metrics],
                                "anomaly_detected": is_anomaly,
                                "anomaly_score": anomaly_score,
                                "diagnosis": diagnosis
                            }
                        )
                        
                        self.active_incidents[incident_id] = incident
                        self.logger.warning(f"Incident detected: {incident_id}")
                        
                        # Attempt to heal
                        await self._heal_incident(incident)
                
                elif degraded_metrics:
                    self.logger.warning(f"Degraded performance detected in {component}: {degraded_metrics}")
                
            except Exception as e:
                self.logger.error(f"Error monitoring {monitor.get_component_name()}: {e}")
    
    async def _heal_incident(self, incident: Incident):
        """Attempt to heal an incident"""
        try:
            diagnosis = incident.metadata.get("diagnosis", {})
            recommended_actions = diagnosis.get("recommended_actions", [])
            
            # Get learning-based recommendation
            learned_fix = self.learning_engine.get_recommended_fix(
                incident.component, 
                incident.description
            )
            
            if learned_fix and learned_fix not in recommended_actions:
                recommended_actions.insert(0, learned_fix)  # Prioritize learned fix
            
            # Try each recommended action
            for action in recommended_actions:
                self.logger.info(f"Attempting fix: {action} for incident {incident.id}")
                
                # Apply fix
                fix_result = await self.autofix_engine.apply_fix(action, {
                    "incident": incident,
                    "component": incident.component,
                    "service_name": incident.component
                })
                
                if fix_result["success"]:
                    incident.fix_applied = action
                    self.logger.info(f"Fix applied successfully: {action}")
                    
                    # Wait a bit for the fix to take effect
                    await asyncio.sleep(10)
                    
                    # Verify the fix
                    verification_result = await self.verification_engine.verify_fix(action, {
                        "incident": incident,
                        "component": incident.component
                    })
                    
                    incident.verification_result = verification_result["verified"]
                    
                    if verification_result["verified"]:
                        # Fix successful
                        incident.resolved_at = datetime.now()
                        self.logger.info(f"Incident {incident.id} resolved successfully")
                        
                        # Learn from successful resolution
                        self.learning_engine.learn_from_incident(incident)
                        
                        # Remove from active incidents
                        if incident.id in self.active_incidents:
                            del self.active_incidents[incident.id]
                        
                        break
                    else:
                        self.logger.warning(f"Fix verification failed for {action}")
                else:
                    self.logger.error(f"Fix failed: {fix_result.get('error', 'Unknown error')}")
            
            # If no fix worked, escalate
            if not incident.resolved_at:
                await self._escalate_incident(incident)
                
        except Exception as e:
            self.logger.error(f"Error healing incident {incident.id}: {e}")
            await self._escalate_incident(incident)
    
    async def _escalate_incident(self, incident: Incident):
        """Escalate incident to human operators"""
        self.logger.critical(f"Escalating incident {incident.id} - automatic healing failed")
        
        # In a real implementation, this would:
        # - Send alerts to operations team
        # - Create tickets in incident management system
        # - Trigger emergency procedures
        
        # Learn from failed resolution
        self.learning_engine.learn_from_incident(incident)
    
    async def _train_anomaly_detector(self):
        """Train anomaly detector with historical normal data"""
        # In a real implementation, this would load historical metrics
        # For demo purposes, we'll generate some sample data
        
        historical_data = []
        for _ in range(100):
            # Generate normal system metrics
            cpu = np.random.normal(30, 10)  # Normal CPU around 30%
