class Calculator:
    """Simple calculator for testing."""
    
    def add(self, a: float, b: float) -> float:
        return a + b
    
    def divide(self, a: float, b: float) -> float:
        if b == 0:
            raise ValueError("Division by zero")
        return a / b
    
    def factorial(self, n: int) -> int:
        if n < 0:
            raise ValueError("Factorial of negative number")
        if n == 0 or n == 1:
            return 1
        return n * self.factorial(n - 1)

class TestCalculatorProperties:
    """Property-based tests for calculator."""
    
    def setup_method(self):
        self.calc = Calculator()
    
    @given(st.floats(allow_nan=False, allow_infinity=False))
    def test_add_identity(self, x: float):
        """Adding zero should return the original number."""
        result = self.calc.add(x, 0)
        assert result == x
    
    @given(

        st.floats(allow_nan=False, allow_infinity=False)
    )

class UserRepository:
    """User repository for testing."""
    
    def __init__(self):
        self.users: Dict[int, User] = {}
        self.next_id = 1
    
    def create_user(self, user: User) -> User:
        if user.id in self.users:
            raise ValueError(f"User with id {user.id} already exists")
        
        if user.id <= 0:
            user.id = self.next_id
            self.next_id += 1
        
        self.users[user.id] = user
        return user
    
    def get_user(self, user_id: int) -> Optional[User]:
        return self.users.get(user_id)
    
    def update_user(self, user: User) -> User:
        if user.id not in self.users:
            raise ValueError(f"User with id {user.id} not found")
        self.users[user.id] = user
        return user
    
    def delete_user(self, user_id: int) -> bool:
        if user_id in self.users:
            del self.users[user_id]
            return True
        return False
    
    def list_active_users(self) -> List[User]:
        return [user for user in self.users.values() if user.is_active]

# test_hypothesis_stateful.py
from hypothesis.stateful import RuleBasedStateMachine, rule, invariant, initialize
from hypothesis import strategies as st
import pytest

class BankAccount:
    """Bank account for stateful testing."""
    
    def __init__(self, initial_balance: float = 0.0):
        self.balance = initial_balance
        self.transaction_history = []
    
    def deposit(self, amount: float) -> None:
        if amount <= 0:
            raise ValueError("Deposit amount must be positive")
        self.balance += amount
        self.transaction_history.append(f"DEPOSIT: {amount}")
    
    def withdraw(self, amount: float) -> None:
        if amount <= 0:
            raise ValueError("Withdrawal amount must be positive")
        if amount > self.balance:
            raise ValueError("Insufficient funds")
        self.balance -= amount
        self.transaction_history.append(f"WITHDRAW: {amount}")
    
    def get_balance(self) -> float:
        return self.balance
    
    def get_transaction_count(self) -> int:
        return len(self.transaction_history)

class BankAccountStateMachine(RuleBasedStateMachine):
    """Stateful testing for BankAccount."""
    
    def __init__(self):
        super().__init__()
        self.account = None
    
    @initialize()
    def setup_account(self):
        """Initialize the bank account."""
        initial_balance = 1000.0
        self.account = BankAccount(initial_balance)
        self.expected_balance = initial_balance
        self.transaction_count = 0
    
    @rule(amount=st.floats(min_value=0.01, max_value=1000.0))
    def deposit(self, amount: float):
        """Rule for depositing money."""
        self.account.deposit(amount)
        self.expected_balance += amount
        self.transaction_count += 1
    
    @rule(amount=st.floats(min_value=0.01, max_value=500.0))
    def withdraw(self, amount: float):
        """Rule for withdrawing money."""
        if amount <= self.account.get_balance():
            self.account.withdraw(amount)
            self.expected_balance -= amount
            self.transaction_count += 1
    
    @invariant()
    def balance_is_consistent(self):
        """Invariant: balance should match expected balance."""
        assert abs(self.account.get_balance() - self.expected_balance) < 0.01
    
    @invariant()
    def balance_is_non_negative(self):
        """Invariant: balance should never be negative."""
        assert self.account.get_balance() >= 0
    
    @invariant()
    def transaction_count_is_consistent(self):
        """Invariant: transaction count should match history."""
        assert self.account.get_transaction_count() == self.transaction_count

class StringProcessor:
    """String processing utilities for mutation testing."""
    
    @staticmethod
    def is_palindrome(s: str) -> bool:
        """Check if string is a palindrome."""
        if not s:
            return True
        
        cleaned = ''.join(c.lower() for c in s if c.isalnum())
        return cleaned == cleaned[::-1]
    
    @staticmethod
    def count_vowels(s: str) -> int:
        """Count vowels in string."""
        if not s:
            return 0
        
        vowels = 'aeiouAEIOU'
        count = 0
        for char in s:
            if char in vowels:
                count += 1
        return count
    
    @staticmethod
    def reverse_words(s: str) -> str:
        """Reverse words in string."""
        if not s:
            return s
        
        words = s.split()
        return ' '.join(reversed(words))
    
    @staticmethod
    def capitalize_words(s: str) -> str:
        """Capitalize first letter of each word."""
        if not s:
            return s
        
        result = []
        words = s.split()
        for word in words:
            if word:
                result.append(word[0].upper() + word[1:].lower())
        return ' '.join(result)

class MathOperations:
    """Math operations for mutation testing."""
    
    @staticmethod
    def factorial(n: int) -> int:
        """Calculate factorial."""
        if n < 0:
            raise ValueError("Factorial not defined for negative numbers")
        if n == 0 or n == 1:
            return 1
        
        result = 1
        for i in range(2, n + 1):
            result *= i
        return result
    
    @staticmethod
    def fibonacci(n: int) -> int:
        """Calculate nth Fibonacci number."""
        if n < 0:
            raise ValueError("Fibonacci not defined for negative numbers")
        if n == 0:
            return 0
        if n == 1:
            return 1
        
        a, b = 0, 1
        for _ in range(2, n + 1):
            a, b = b, a + b
        return b
    
    @staticmethod
    def is_prime(n: int) -> bool:
        """Check if number is prime."""
        if n < 2:
            return False
        if n == 2:
            return True
        if n % 2 == 0:
            return False
        
        for i in range(3, int(n ** 0.5) + 1, 2):
            if n % i == 0:
                return False
        return True
    
    @staticmethod
    def gcd(a: int, b: int) -> int:
        """Calculate greatest common divisor."""
        a, b = abs(a), abs(b)
        while b:
            a, b = b, a % b
        return a
