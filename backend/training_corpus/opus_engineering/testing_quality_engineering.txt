## 1. Property-Based Testing with Hypothesis {#property-based-testing}

Property-based testing generates test cases automatically based on specifications of what the code should do, rather than specific examples.

### Core Concepts and Implementation

```python

# src/math_utils.py
from typing import List, Union
import math

def sort_list(items: List[int]) -> List[int]:
    """Sort a list of integers."""
    return sorted(items)

def reverse_string(s: str) -> str:
    """Reverse a string."""
    return s[::-1]

def calculate_mean(numbers: List[Union[int, float]]) -> float:
    """Calculate arithmetic mean of numbers."""
    if not numbers:
        raise ValueError("Cannot calculate mean of empty list")
    return sum(numbers) / len(numbers)

def fibonacci(n: int) -> int:
    """Calculate nth Fibonacci number."""
    if n < 0:
        raise ValueError("n must be non-negative")
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

def gcd(a: int, b: int) -> int:
    """Calculate greatest common divisor."""
    while b:
        a, b = b, a % b
    return abs(a)

class BankAccount:
    def __init__(self, initial_balance: float = 0.0):
        if initial_balance < 0:
            raise ValueError("Initial balance cannot be negative")
        self._balance = initial_balance
    
    def deposit(self, amount: float) -> None:
        if amount <= 0:
            raise ValueError("Deposit amount must be positive")
        self._balance += amount
    
    def withdraw(self, amount: float) -> None:
        if amount <= 0:
            raise ValueError("Withdrawal amount must be positive")
        if amount > self._balance:
            raise ValueError("Insufficient funds")
        self._balance -= amount
    
    @property
    def balance(self) -> float:
        return self._balance
```

            assert fib_n == 0

            assert fib_n == 1

        assert result > 0

        assert gcd(a, b) == gcd(b, a)

class BankAccountStateMachine(RuleBasedStateMachine):
    """Stateful testing for BankAccount class."""
    
    def __init__(self):
        super().__init__()
        self.account = BankAccount(100.0)  # Start with $100

        self.account.deposit(amount)
        self.expected_balance += amount
    
    @rule(amount=st.floats(min_value=0.01, max_value=50.0))
    def withdraw(self, amount: float):
        """Rule for withdrawing money."""
        assume(amount <= self.expected_balance)
        note(f"Withdrawing ${amount:.2f}")
        self.account.withdraw(amount)
        self.expected_balance -= amount
    
    @invariant()
    def balance_is_correct(self):
        """Invariant: balance should match expected balance."""
        assert abs(self.account.balance - self.expected_balance) < 0.01
    
    @invariant()
    def balance_is_non_negative(self):
        """Invariant: balance should never be negative."""
        assert self.account.balance >= 0

## 2. Mutation Testing Strategies {#mutation-testing}

Mutation testing evaluates test quality by introducing bugs and checking if tests catch them.

```python
# requirements.txt additions

# src/calculator.py
from typing import Union, List
import math

class Calculator:
    """A calculator class for mutation testing demonstration."""
    
    def add(self, a: Union[int, float], b: Union[int, float]) -> Union[int, float]:
        """Add two numbers."""
        return a + b
    
    def subtract(self, a: Union[int, float], b: Union[int, float]) -> Union[int, float]:
        """Subtract b from a."""
        return a - b
    
    def multiply(self, a: Union[int, float], b: Union[int, float]) -> Union[int, float]:
        """Multiply two numbers."""
        return a * b
    
    def divide(self, a: Union[int, float], b: Union[int, float]) -> float:
        """Divide a by b."""
        if b == 0:
            raise ZeroDivisionError("Cannot divide by zero")
        return a / b
    
    def power(self, base: Union[int, float], exponent: Union[int, float]) -> Union[int, float]:
        """Raise base to the power of exponent."""
        return base ** exponent
    
    def square_root(self, x: Union[int, float]) -> float:
        """Calculate square root of x."""
        if x < 0:
            raise ValueError("Cannot calculate square root of negative number")
        return math.sqrt(x)
    
    def factorial(self, n: int) -> int:
        """Calculate factorial of n."""
        if n < 0:
            raise ValueError("Factorial is not defined for negative numbers")
        if n == 0 or n == 1:
            return 1
        result = 1
        for i in range(2, n + 1):
            result *= i
        return result
    
    def is_prime(self, n: int) -> bool:
        """Check if n is a prime number."""
        if n < 2:
            return False
        if n == 2:
            return True
        if n % 2 == 0:
            return False
        
        for i in range(3, int(math.sqrt(n)) + 1, 2):
            if n % i == 0:
                return False
        return True
    
    def gcd(self, a: int, b: int) -> int:
        """Calculate greatest common divisor using Euclidean algorithm."""
        while b:
            a, b = b, a % b
        return abs(a)
    
    def lcm(self, a: int, b: int) -> int:
        """Calculate least common multiple."""
        return abs(a * b) // self.gcd(a, b)

def process_list(numbers: List[int]) -> dict:
    """Process a list of numbers and return statistics."""
    if not numbers:
        return {
            'sum': 0,
            'average': 0,
            'min': None,
            'max': None,
            'count': 0
        }
    
    total = sum(numbers)
    count = len(numbers)
    
    return {
        'sum': total,
        'average': total / count,
        'min': min(numbers),
        'max': max(numbers),
        'count': count
    }

def find_max_subarray_sum(arr: List[int]) -> int:
    """Find maximum sum of contiguous subarray (Kadane's algorithm)."""
    if not arr:
        return 0
    
    max_sum = current_sum = arr[0]
    
    for i in range(1, len(arr)):
        current_sum = max(arr[i], current_sum + arr[i])
        max_sum = max(max_sum, current_sum)
    
    return max_sum

def binary_search(arr: List[int], target: int) -> int:
    """Binary search implementation. Returns index or -1 if not found."""
    left, right = 0, len(arr) - 1
    
    while left <= right:
        mid = (left + right) // 2
        
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    return -1
```

    def test_empty_array(self):
        assert binary_search([], 5) == -1
    
    def test_single_element_found(self):
        assert binary_search([5], 5) == 0
    
    def test_single_element_not_found(self):
        assert binary_search([5], 3) == -1
    
    def test_multiple_elements_found(self):
        arr = [1, 3, 5, 7, 9, 11]
        assert binary_search(arr, 1) == 0  # First element
        assert binary_search(arr, 11) == 5  # Last element
        assert binary_search(arr, 5) == 2  # Middle element
        assert binary_search(arr, 7) == 3  # Another middle element
    
    def test_multiple_elements_not_found(self):
        arr = [1, 3, 5, 7, 9, 11]
        assert binary_search(arr, 0) == -1  # Less than first
        assert binary_search(arr, 12) == -1  # Greater than last
        assert binary_search(arr, 4) == -1  # Between elements
        assert binary_search(arr, 6) == -1  # Between elements
    
    def test_boundary_conditions(self):
        arr = [2, 4, 6, 8, 10]
        # Test conditions that might be mutated (<=, <, >=, >)
        assert binary_search(arr, 2) == 0
        assert binary_search(arr, 10) == 4
        assert binary_search(arr, 1) == -1
        assert binary_search(arr, 11) == -1
