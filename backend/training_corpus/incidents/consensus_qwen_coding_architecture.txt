class MorphogenesisLoop:
    """Grace's self-development loop - she builds herself iteratively"""
    
    def __init__(self):
        self.max_depth = 100  # Circuit breaker - prevents infinite recursion
        self.current_depth = 0
        self.qwen_agent = QwenCodingAgent()
        self.memory_bridge = MemoryBridge()
        self.token_monitor = TokenMonitor()
        self.consensus_chat = ConsensusChat()
        
    def execute(self, user_request: str):
        """Main loop: User request → Blueprint → Code → Integration"""
        
        # Phase 1: Consensus Design
        blueprint = self.consensus_chat.design_feature(user_request)
        
        # Phase 2: Chunked Implementation
        chunks = self.chunk_blueprint(blueprint)
        
        for chunk in chunks:
            self.implement_chunk(chunk)
            
    def implement_chunk(self, chunk):
        """Implement one chunk with token management"""
        
        if self.current_depth >= self.max_depth:
            raise CircuitBreakerException("Max depth reached")
            
        # Load context for this chunk
        context = self.memory_bridge.load_context_for_chunk(chunk)
        
        # Monitor tokens while Qwen works
        with self.token_monitor.watch():
            code = self.qwen_agent.generate_code(chunk, context)
            
        # Save results and update memory
        self.memory_bridge.save_chunk_result(chunk, code)
        self.current_depth += 1

Grace Self-Building Starter Kit
Run this file → it creates the first module (grace/memory_bridge.py)

No external deps except open-webui (or any OpenAI-compatible endpoint).

import os, json, time, hashlib, requests
from typing import List, Dict

# ---------- 1-line config ----------
API_URL     = "http://localhost:11434/v1/chat/completions"  # Ollama, OpenWebUI, etc.
MODEL_NAME  = "qwen2.5-coder:7b"
MAX_TOKENS  = 3500           # real input budget
SAFETY      =  500           # leave this head-room
CHUNK_TOKENS= 1200           # max size of one work-unit
MAX_DEPTH   = 100            # circuit-breaker
MEMORY_DIR  = ".grace/memory"
os.makedirs(MEMORY_DIR, exist_ok=True)

# ---------- helpers ----------
def tok(x:str)->int: return len(x.encode())//4        # fast token estimator
def sha256(x:str)->str: return hashlib.sha256(x.encode()).hexdigest()[:16]
def now(): return int(time.time())

# ---------- memory bridge ----------
class MemoryBridge:
    def __init__(self):
        self.index_file = os.path.join(MEMORY_DIR, "index.json")
        self.index = json.load(open(self.index_file)) if os.path.exists(self.index_file) else {}

    def commit(self, key:str, artefact:str, meta:dict):
        path = os.path.join(MEMORY_DIR, f"{key}.py")
        open(path,"w").write(artefact)
        self.index[key] = {"t":now(), "meta":meta}
        json.dump(self.index, open(self.index_file,"w"))

    def selective_memories(self, chunk_desc:str, budget:int)->str:
        "Returns ≤ budget tokens of most relevant past artefacts."
        scored = []
        for k,v in self.index.items():
            score  = 50*(v["meta"].get("success",0))        # 0 or 1
            score += 30/(1+ (now()-v["t"])/3600 )           # age
            score += 10*(v["meta"].get("fail",0))           # past failures
            scored.append((score, k))
        scored.sort(reverse=True)
        out, used = [], 0
        for _,k in scored:
            txt = open(os.path.join(MEMORY_DIR,f"{k}.py")).read()
            if used + tok(txt) > budget: break
            out.append(txt); used+=tok(txt)
        return "\n# --- memory ---\n".join(out)

# ---------- Qwen caller ----------
def qwen(prompt:str)->str:
    resp = requests.post(API_URL, json={
        "model": MODEL_NAME,
        "messages":[{"role":"user","content":prompt}],
        "temperature":0.15,
        "max_tokens":SAFETY
    }, timeout=120)
    return resp.json()["choices"][0]["message"]["content"]

# ---------- main loop ----------
def build(user_story:str):
    bridge = MemoryBridge()
    genesis = sha256(user_story)
    depth   = 0
    while depth < MAX_DEPTH:
        depth += 1
        print(f"\n[ITER {depth}] ----------------------------------")
        # --- 1. blueprint chunk ---
        blueprint = f"Write ONE python module that satisfies: {user_story}"
        chunk     = "module code (≤1200 tokens)"
        # --- 2. build context ---
        memories  = bridge.selective_memories(chunk, budget=1000)
        compass   = "# Architecture compass: modules talk via MemoryBridge only\n"
        playbook  = "# Playbook: small pure functions, immutable artefacts, sha256 keys\n"
        prompt    = f"{compass}{playbook}\n{memories}\n\nTASK: {chunk}\n```python\n"
        if tok(prompt) > MAX_TOKENS:
            print("Prompt too fat – trimming memories.")
            memories = bridge.selective_memories(chunk, budget=500)
            prompt   = f"{compass}{playbook}\n{memories}\n\nTASK: {chunk}\n```python\n"
        # --- 3. code ---
        code = qwen(prompt).removeprefix("```python").removesuffix("```").strip()
        print("Generated:\n", code[:500]+"..." if len(code)>500 else code)
        # --- 4. test & feedback ---
        try:
            compile(code, f"<genesis-{genesis}-d{depth}>", "exec")
            success = True; fail = False
            print("✅ compiles")
        except SyntaxError as e:
            success = False; fail = True
            print("❌ compile error – saving anyway for post-mortem")
        # --- 5. store ---
        bridge.commit(f"{genesis}-d{depth}", code,
                      {"story":user_story, "chunk":chunk, "success":success, "fail":fail})
        # --- 6. circuit-breaker ---
        if success and tok(code) < CHUNK_TOKENS:
            print("Module complete – stop.")
            break

# ---------- run ----------
if __name__ == "__main__":
    story = input("What should Grace build next?\n> ")
    build(story)
