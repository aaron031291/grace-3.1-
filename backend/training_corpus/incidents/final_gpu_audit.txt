```python
import json, os, datetime, re, requests, subprocess, duckduckgo_search as ddg

FAIL_DIR   = os.path.expanduser("~/.grace/failures")
KNOW_DIR   = os.path.expanduser("~/.grace/knowledge")
PROC_DIR   = os.path.expanduser("~/.grace/procedures")
LOOP_MAX   = 6               # human escalation after 6
LOCAL_LLM  = "http://127.0.0.1:8000/v1/chat/completions"

def log(tag, data):
    ts = datetime.datetime.utcnow().isoformat(timespec="seconds")
    with open(f"{FAIL_DIR}/{tag}_{ts}.json", "w") as f:
        json.dump(data, f, indent=2)

def search_web(q, n=3):
    return [r["body"] for r in ddg.DDGS().text(q, max_results=n)]

def ask_local_llm(prompt):
    r = requests.post(LOCAL_LLM, json={"model":"codellama","messages":[{"role":"user","content":prompt}],"temperature":0}, timeout=60)
    return r.json()["choices"][0]["message"]["content"]

def teach_me(concept):
    tutor_prompt = f"Explain the concept '{concept}' to an AI repair daemon. Keep under 500 tokens, include a minimal runnable Python example."
    return ask_local_llm(tutor_prompt)

def store_procedure(tag, content):
    open(f"{PROC_DIR}/{tag}.md","w").write(content)

def recursive_retry(*, task, code, error, attempt=1):
    tag = re.sub(r"\W+","_", task)[:40]
    log(tag, {"task":task, "code":code, "error":str(error), "attempt":attempt})

    if attempt > LOOP_MAX:
        raise RuntimeError("Governance escalation – human required.")

    # 1. extract missing concept
    concept = ask_local_llm(f"Extract the single Python or system concept this error is about:\n{error}\nReply with one noun phrase only.")
    # 2. learn it
    snippets = search_web(concept+" python tutorial", 3)
    explanation = teach_me(concept)
    store_procedure(f"{tag}_a{attempt}", explanation)
    # 3. patch code
    new_code = ask_local_llm(f"Fix this code using the concept above:\n{code}\nError: {error}\nReturn only the corrected Python.")
    log(tag+"_retry", {"attempt":attempt, "concept":concept, "new_code":new_code})
    # 4. try again
    try:
        exec(new_code, {"__builtins__":__builtins__})
        return new_code                     # success
    except Exception as e:
        return recursive_retry(task=task, code=new_code, error=e, attempt=attempt+1)
```
